const Find_Digits = (N) => 
{
	 let count = 0;
	 while(N > 0){
	   N = Math.floor(N/10);
	   count++;
	 }
	 return count;
};

const Find_Five = (N) => 
{
	 let count = 0;
	 while(N > 0) {
	   let a = N%10;
	   if(a === 5) {
	     count++;
	   }
	   N = Math.floor(N/10);
	 }
  return count;
};

{

Que7: Check if the number is Prime or not

Ans:
const Prime_Check = (N) => 
{
  let div = 0;
	 for(let i = 1; i <= N; i++){
	   if(N % i === 0){
	     div++;
	   }
	 }
	if(div === 2){
	  return "YES";
	}
	else {
	  return "NO";
	}
};
